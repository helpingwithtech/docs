export const Button = ({ 
  children, 
  intent = "primary", 
  size = "medium", 
  className = "", 
  href,
  ...props 
}) => {
  const baseStyles = "inline-flex items-center justify-center font-medium transition-colors";
  
  const intentStyles = {
    primary: "bg-[#f7c133] hover:bg-[#f2b700] text-white border-2 border-zinc-900",
    secondary: "bg-[#1B1C1E] hover:bg-[#f2b700] text-white border-2 border-zinc-900",
    outline: "bg-white hover:bg-zinc-200 border border-zinc-900 text-zinc-900",
  };
  
  const sizeStyles = {
    small: "px-3 py-1.5 rounded-md text-sm",
    medium: "p-3 rounded-lg text-base",
  };
  
  const combinedClasses = `${baseStyles} ${intentStyles[intent] || intentStyles.primary} ${sizeStyles[size] || sizeStyles.medium} ${className}`;
  
  if (href) {
    return (
      <a href={href} className={combinedClasses} {...props}>
        {children}
      </a>
    );
  }
  
  return (
    <button className={combinedClasses} {...props}>
      {children}
    </button>
  );
};
