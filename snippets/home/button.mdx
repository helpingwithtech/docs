export const Button = ({ 
  children, 
  intent = "primary", 
  size = "medium", 
  className = "", 
  href,
  ...props 
}) => {
  const baseStyles = "inline-flex items-center justify-center font-medium transition-all duration-200";
  
  const intentClasses = {
    primary: "border-2",
    secondary: "border-2",
    outline: "border",
  };
  
  const intentInlineStyles = {
    primary: { backgroundColor: "#f7c133", color: "#ffffff", borderColor: "#18181b" },
    secondary: { backgroundColor: "#1B1C1E", color: "#ffffff", borderColor: "#18181b" },
    outline: { backgroundColor: "#ffffff", color: "#18181b", borderColor: "#18181b" },
  };
  
  const sizeStyles = {
    small: "px-4 py-2 rounded-md text-sm",
    medium: "px-5 py-3 rounded-lg text-base",
  };
  
  const combinedClasses = `${baseStyles} ${intentClasses[intent] || intentClasses.primary} ${sizeStyles[size] || sizeStyles.medium} ${className}`;
  const inlineStyle = intentInlineStyles[intent] || intentInlineStyles.primary;
  
  if (href) {
    return (
      <a 
        href={href} 
        className={combinedClasses} 
        style={inlineStyle}
        onMouseEnter={(e) => {
          if (intent === "primary") e.currentTarget.style.backgroundColor = "#f2b700";
          if (intent === "secondary") e.currentTarget.style.backgroundColor = "#f2b700";
          if (intent === "outline") e.currentTarget.style.backgroundColor = "#e4e4e7";
        }}
        onMouseLeave={(e) => {
          e.currentTarget.style.backgroundColor = inlineStyle.backgroundColor;
        }}
        {...props}
      >
        {children}
      </a>
    );
  }
  
  return (
    <button 
      className={combinedClasses} 
      style={inlineStyle}
      onMouseEnter={(e) => {
        if (intent === "primary") e.currentTarget.style.backgroundColor = "#f2b700";
        if (intent === "secondary") e.currentTarget.style.backgroundColor = "#f2b700";
        if (intent === "outline") e.currentTarget.style.backgroundColor = "#e4e4e7";
      }}
      onMouseLeave={(e) => {
        e.currentTarget.style.backgroundColor = inlineStyle.backgroundColor;
      }}
      {...props}
    >
      {children}
    </button>
  );
};
